  1. What is the output given these four numbers as input: 5  1  -6  3 
       int n;
   while (cin >> n) {
       for (int i=0; i<n; i++) {
           cout << i << " ";
       }
       cout << endl;
   }
  2. What is the output from the following code, given these two numbers as input: 5  4? 
       int rows, cols;
   cin >> rows >> cols;
   for (int r=1; r<=rows; r++) {
       for (int c=1; c<=cols; c++) {
           if (r==1 || r==rows || c==1 || c==cols) {
               cout << 'X';
           } else {
               cout << '-';
           }
       }
       cout << endl;
   }
  3. What is the "while" loop equivalent of this "for" loop? 
        int n;
    cin >> n;
    int sum = 0;
    for (int i=0; i<n; i++) {
        sum = sum + i;
    }
  4. What is the "for" loop equvalent of this "while" loop? 
        int i = 25;
    while (i > 0) {
        if (i%2 == 1) {
            cout << i << endl;
        }
       i--;
    }
