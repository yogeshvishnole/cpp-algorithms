  5. What is the output given input: 5 3 -6 8 -2 1 
       // Loop written in "Sentinal" style.
   int number;
   int sum = 0;
   bool nonNegative = true;
   bool notEOF = true;
   while (nonNegative && notEOF) {
       if (cin >> number) {
           if (number < 0) {
               nonNegative = false;
           } else {
               sum = sum + number;
           }
      }else{
           notEOF = false;
      }
   }
   cout << sum << endl;
  6. What is the output given input: 5 3 -6 8 -2 1

       sum += number;
