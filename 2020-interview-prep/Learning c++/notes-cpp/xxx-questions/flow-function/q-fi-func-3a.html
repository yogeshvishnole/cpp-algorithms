int nc(int n, char c){
    for (int i=0; i<n; i++) {
        cout << c;
    }
    return n+1;
}

void xy(int m){
    cout << '-';
    if (m%2 == 0) {
        nc(m/2, 'x');
    } else {
        nc(3*m+1, 'y');
    }
    cout << '-';
}

  2. __________ nc(nc(2, 'c'), 'd');
