// arrays/sort-students.cpp - sort students by score
// Fred Swartz - 2003-09-24

#include <iostream>
#include <string>
using namespace std;

//========================================= type definition
struct Student {
          string first_name;
          string last_name;
          int    score;
       };
       
//============================================== prototypes
void selectionSort(Student x[], int n);

//==================================================== main
int main() {
    Student students[1000];
    int n=0;  // number of students.
    
    //--- Read student info
    while (cin >> students[n].score 
               >> students[n].first_name 
               >> students[n].last_name) {
        n++;
    }
    
    selectionSort(students, n);
    
    //--- Write student info.
    for (int i=0; i<n; i++) {
        cout << students[i].score      << " "
             << students[i].first_name << " "
             << students[i].last_name  << endl;
    }
    
    return 0;
}//end main

//=================================================== selectionSort
void selectionSort(Student x[], int n) {

    for (int pass=0; pass<n-1; pass++) {
        int potentialSmallest = pass;  // assume this is smallest
        //--- Look over remaining elements to find smallest.
        for (int i=pass+1; i<n; i++) {
            if (x[i].score < x[potentialSmallest].score) {
                //--- Remember index for latter swap.
                potentialSmallest = i;
            }
        }
        
        //--- Swap smallest remaining element
        Student temp = x[pass];
        x[pass] = x[potentialSmallest];
        x[potentialSmallest] = temp;
    }
}//end selectionSort
