// floatVector.h - Illustrates typical vector features.
//  Fred Swartz - 2001-11-13 ... 2004-02-17
#include <iostream>
using namespace std;

#ifndef floatVector_H
#define floatVector_H

class floatVector {
    friend ostream& operator<<(ostream& os, const floatVector& fv);
    
  public:
    floatVector();        // default constructor
    floatVector(int cap); // constructor specifying capacity
    floatVector(const floatVector& fv); // copy constructor
    ~floatVector();       // destructor
       
    float& at(int position) const; // ref to element at position  
    float& back() const;           // Value of last elem
    int    capacity() const;       // Max size before reallocation.
    void   clear();                // Delete all items.
    bool   empty() const;          // True if no elements, size()==0.
    void   erase(int position);    // Delete item at position.
    void   insert(int position, float item); // Insert item at position.
    float& front() const;          // Returns first element.
    void   pop_back();             // Remove last element.
    void   push_back(float item);  // Add item to end.
    void   reserve(int cap);       // Insure capacity at least cap.
    int    size() const;           // Number of elements in vector.
    
    //-- Overloaded operators
    float& operator[](int position) const; // Ref to element at position.
    bool   operator==(floatVector& v) const; // true if vectors equal.
    floatVector& operator=(const floatVector& fv);

    //-- Forward iterator (no reverse iterator - yet)
    typedef float* iterator;  // The iterators are simply pointers.
    iterator begin();         // Iterator of first element.
    iterator end();           // Iterator PAST last element.

  private:
    int    _capacity;   // Current maximum size (number of floats).
    int    _size;       // Current "size"
    float* _data;       // Pointer to array of floats
    
    void   _expand();   // Allocate more memory.
    void   _ensure();   // Guarantee at least one empty slot.
};
#endif
