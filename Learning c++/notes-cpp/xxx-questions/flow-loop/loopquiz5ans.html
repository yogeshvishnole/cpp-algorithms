  1. What is the "while" loop equivalent of this "for" loop? 
        int n;
    cin >> n;
    int sum = 0;
    for (int i=0; i<n; i++) {
        sum += i;
    }
    Answer: 
        int n;
    cin >> n;
    int sum = 0;
    int i = 0;
    while (i < n) {
        sum += i;
        i++;
    }

  3. Write a loop to read integers and print the maximum value.
    Answer: 
        int max;           // var to remember maximum value.
    int x;             // var for each value read.
    cin >> max;        // start max with the first value.
    while (cin >> x) { // read each of the remaining numbers
        if (x > max) {
            max = x;   // save this bigger number
        }
    }
    cout << "Maximum = " << max << endl;
  4. Read one number from the input, and write a loop that computes how many integers from 1 to 1,000,000 can be evenly divided by that number. Print the result.
    Answer: 
        int count = 0;      // to count even divisions.
    int divisor;        // num to divide all numbers by.
    cin >> divisor;     // read in the value.
    for (int i=1, i<=1000000; i++) {
        if (i%divisor == 0) {
            count++;    // count this even division
        }
    }
    cout << "Count = " << count << endl;
