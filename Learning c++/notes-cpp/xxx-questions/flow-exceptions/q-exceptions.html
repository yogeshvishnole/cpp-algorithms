Exceptions

The following program uses try-throw-catch. Note that if there is no catch for a type of exception, an error message like "abnormal program termination" is printed and the program is terminated. 
#include <iostream>
#include <stdexcept>
using namespace std;

void f(int x);  // prototype

//=============================== main
int main() {
    int test;
    while (cin >> test) {
        try {
            cout << "1 ";
            f(test);
            cout << "2 ";
        }catch (out_of_range e) {
            cout << "3 ";
            cout << e.what();
            cout << "4 ";
        }
        cout << "5 " << endl;
    }
    return 0;
}//end main


//=============================== f
void f(int x) {
    cout << "6 ";
    if (x < 0) {
        throw out_of_range("7 ");
    }else if (x > 100) {
        throw length_error("8 ");
    }
    cout << "9 ";
}//end f
