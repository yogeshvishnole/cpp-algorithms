#include <iostream>
using namespace std;

//==================================================== prototypes
bool readRange(int& x, int low, int high);

//========================================================== main
int main() {
    int hours, minutes, seconds;
    while (readRange(hours  , 0, 1000) 
        && readRange(minutes, 0,   59) 
        && readRange(seconds, 0,   59)) {
        
        //-- we don't really do anything, just print time
        cout << hours   << " hrs, " 
             << minutes << " mins, "
             << seconds << " secs" << endl << endl;
    }
    return 0;
}//end main


//====================================================== readRange
// Utility function to read int in given range.
bool readRange(int& x, int low, int high) {
    cout << "Enter an integer in the range " 
         << low << " to " << high << endl;
    while (cin >> x) {
        if (x >= low && x <= high) {
            return true; //>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
        }
        cout << x << " is out of range. Reenter." << endl;
    }
    return false;  // EOF or bad input
}//end readRange
