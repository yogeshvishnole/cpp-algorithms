template<class T>
class my_vector {
  public:
    my_vector<T>();        // default constructor
    my_vector<T>(int cap); // constructor specifying capacity
    my_vector<T>(const my_vector& fv); // copy constructor
    ~my_vector<T>();       // destructor

    T&    at(int position) const;      // return element at position
    int   capacity() const;        // current max possible
    void  clear();                 // delete all items
    bool  empty() const;           // true if no elements in vector
    void  erase(int position);     // delete item at position
    void  insert(int position , T item);  // insert item at position
    T&    operator[](int subscript) const;// same as at.
    bool  operator==(my_vector& v) const; // true if contain equal vals
    void  pop_back();              // return last element and remove
    void  push_back(T item);       // add item to end
    void  replace(int position, T item); // replace item at position
    void  resize(int new_capacity);// reallocate memory
    int   size() const;            // number of elements in vector

  private:
    int  _capacity; // current allocation for _info array.
    int  _size;     // current "size"
    T *  _info;     // pointer to array of Ts
};
