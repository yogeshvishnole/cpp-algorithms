       float xyz[] = {12.5, 1.3333, 8.1, 7.0};
   float zz[5] = {33.3, 9.0, 234.5, 25.25, 14.9};
   . . .

   cout << indexOf(3.3 , xyz, 4); // prints -1 ie, not found
